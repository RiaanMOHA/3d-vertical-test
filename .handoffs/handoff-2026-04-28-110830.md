### Chrome headless WebGL on macOS

`--use-angle=metal --virtual-time-budget=8000` works for 3D rendering screenshots. Other variants (`--use-gl=swiftshader`, `--use-angle=gl`) gave warnings or black canvases. `--virtual-time-budget` is essential — without it the page loads but doesn't have time to render the WebGL context.

### Big inlined HTML files: line-based tooling

When a single file has a huge inlined library on one line (the Three.js r128 case here = 603KB on one line), normal file-reading tools fail. Use:

```bash
awk 'BEGIN{}{LC++; if (length($0)>500) print LC, length($0)}' file       # find oversized lines
awk 'NR>=A && NR<=B' file                                                  # extract a line range
awk 'NR!=51' file | grep -nE '...'                                         # grep skipping the giant line
sed -n 'X,Yp' file                                                          # works fine
```

Bash output gets persisted-to-file when too large, but `awk 'NR!=51'` keeps it manageable.

### Scene registry pattern in ozu-test.html

```javascript
const scenes = {};
function registerScene(id, cfg) {
  const group = new THREE.Group();
  group.visible = false;
  scene.add(group);
  const S = { id, group, walls: [], rooms: cfg.rooms || [],
              orbitTarget: cfg.orbitTarget, orbitPos: cfg.orbitPos,
              fpStart: cfg.fpStart, fpYaw: cfg.fpYaw || 0,
              bg: cfg.bg || 0xeef2f6, build: cfg.build };
  scenes[id] = S;
  cfg.build(S);  // builds geometry into S.group
}

function switchScene(id) {
  const S = scenes[id]; if (!S) return;
  for (const k of Object.keys(scenes)) scenes[k].group.visible = (k === id);
  activeScene = id;
  orbit.setFrom(S.orbitPos, S.orbitTarget);
  scene.background = new THREE.Color(S.bg);
  if (document.pointerLockElement) document.exitPointerLock();
  for (const chip of document.querySelectorAll('.chip')) {
    chip.classList.toggle('active', chip.dataset.scene === id);
  }
  walkBtn.style.display = S.noFirstPerson ? 'none' : 'inline-block';
}
```

To add another scene later (e.g. `'walk'`, `'cutaway'`, room-specific viewpoints): just call `registerScene('id', { ... })` and add a `<div class="chip" data-scene="id">label</div>` in the chip row.
